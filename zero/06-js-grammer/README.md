
3 补全函数。
```
// 对 传入的所以参数求和
function superAdd() {
  // 你写的内容
}

superAdd(1) // 返回值 1
superAdd(1, 2) // 返回值 3
superAdd(1, 2, 3, 4) // 返回值 10
superAdd(1, 2, 3, 4, 5) // 返回值 15
```
